# MAPE
def MAPE(Y_true, Y_pred):
    Y_true = np.array(Y_true).flatten()
    Y_pred = np.array(Y_pred).flatten()
    return (1/Y_true.shape[0])*np.sum(np.absolute(Y_pred-Y_true)/np.absolute(Y_true))


# Adj-R2
def adj_r2_score(y_true, y_pred, n, p):
    adj_r2 = 1-(1-r2_score(y_true, y_pred))*((n-1)/(n-p-1))
    return adj_r2
